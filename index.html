function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = JSON.parse(e.postData.contents); sheet.appendRow([ new Date(), data.service || '', data.name || '', data.phone || '', data.email || '', data.address || '', data.contactPref || '', data.photoCount || '', data.miles || '', data.estPrice || '', data.prefDate || '', data.prefTime || '', data.comments || '', data.trenchDepth || '' ]); return ContentService.createTextOutput(JSON.stringify({status: 'success'})) .setMimeType(ContentService.MimeType.JSON); }
